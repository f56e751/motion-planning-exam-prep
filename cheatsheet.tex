\sect{1. C-space \& Completeness \hfill {\scriptsize[L00,02]}}
$\mathcal{C}_{\text{obs}} = \{q : A(q)\cap\mathcal{O}\neq\emptyset\}$, $\mathcal{C}_{\text{free}}=\mathcal{C}\setminus\mathcal{C}_{\text{obs}}$. Plan: continuous $\tau:[0,1]\to\mathcal{C}_{\text{free}}$.
\begin{itemize}
\item \sub{Complete}: returns path iff exists.
\item \sub{Resolution complete}: at chosen discretization.
\item \sub{Probabilistic complete}: $P(\text{fail})\to 0$ as $n\to\infty$.
\end{itemize}
Minkowski sum: $\mathcal{C}_{\text{obs}} = \mathcal{O}\oplus A$ for point robot.

\sect{2. Roadmaps \hfill {\scriptsize[L00]}}
\sub{Visibility graph}: nodes = obstacle vertices + start/goal. Edge if segment is obstacle edge or doesn't cross. $O(n^3)$ naive, $O(n^2)$ optimal.

\sub{Voronoi diagram}: $\{q : |\text{near}(q)|>1\}$ — maximally clears obstacles. Straight arc (edge-edge / vertex-vertex), parabolic (edge-vertex). Retraction $\rho:\mathcal{C}_{\text{free}}\to\text{Vor}$.

\sub{Cell decomposition}: trapezoidal (exact, complete), quadtree/octree (approximate). Build adjacency graph, search.

\sect{3. Potential Field \hfill {\scriptsize[L00]}}
$U_{\text{att}}=\tfrac12 k_a\|x-x_g\|^2$, $F_{\text{att}}=-k_a(x-x_g)$.
$F_{\text{rep}}=k_r(\tfrac1\rho-\tfrac1{\rho_0})\tfrac1{\rho^2}\nabla\rho$ if $\rho\le\rho_0$.
Issue: local minima. Fix: random walks (RPP), navigation function (no local min, $\infty$ at obstacles, smooth), wave-front (resolution complete, search from goal).

\sect{4. Graph Search \hfill {\scriptsize[L00,01]}}
BFS: $O(b^d)$ time/space, complete, optimal (unweighted). DFS: $O(h)$ space, not optimal.

\sub{Dijkstra}: $f(n)=g(n)$ (no heuristic).

\sub{A*}: $f(n)=g(n)+h(n)$.
\begin{itemize}
\item Optimal iff $h$ \sub{admissible} ($h(n)\le h^*(n)$).
\item \sub{Consistent}: $h(s)\le c(s,s')+h(s')$, $h(\text{goal})=0$.
\item Open list (priority queue), Closed list.
\item Optimally efficient among same-$h$ algorithms.
\end{itemize}

\sub{Best-first / Greedy}: $f=h$. \sub{Beam search}: keep $k$ best per level. \sub{RBFS}: DFS-like with $f$-value backup, $O(bd)$ memory.

\sect{5. D*, LPA*, D* Lite \hfill {\scriptsize[L01,03]}}
\sub{LPA*} (incremental A*, fixed start):
$rhs(s)=\min_{s'\in\text{Pred}(s)}(g(s')+c(s',s))$, $rhs(s_{\text{start}})=0$.
\sub{Locally consistent}: $g=rhs$. Inconsistent in priority queue.
Key $[k_1,k_2]=[\min(g,rhs)+h;\ \min(g,rhs)]$.
$g>rhs$ (overconsistent): set $g=rhs$.
$g<rhs$ (underconsistent): set $g=\infty$.

\sub{D* Lite} (reverse, moving agent):
Search goal$\to$start. $rhs(s)=\min_{s'\in\text{Succ}(s)}(c(s,s')+g(s'))$, $rhs(s_{\text{goal}})=0$.
Key modifier $k_m\mathrel{+}=h(s_{\text{last}},s_{\text{curr}})$ avoids re-sorting queue when robot moves.

\sect{6. PRM (multi-query) \hfill {\scriptsize[L02]}}
\textbf{Construction}: sample $n$ collision-free $q$'s, connect each to $k$ nearest via local planner (e.g. straight line + incremental/subdivision collision check). \textbf{Query}: connect $q_i,q_g$ to roadmap, A*/Dijkstra.
Variants: OBPRM (sample near obstacle surfaces), Lazy PRM.
Distance: $d(q,q')=w_1\|X-X'\|+w_2 f(R,R')$.

\sect{7. RRT \& RRT* \hfill {\scriptsize[L02]}}
\textbf{Build}: $q_{\text{rand}}$ → nearest $q_{\text{near}}$ → step toward → $q_{\text{new}}$. Add if collision-free. Goal bias 5–10\%.
\textbf{Merge} (bidirectional): grow two trees, alternate.
Nodes with largest Voronoi region most likely expanded.

\sub{RRT*}: Within radius of $q_{\text{new}}$:
\begin{enumerate}
\item Choose parent minimizing total cost.
\item \sub{Rewire}: if going through $q_{\text{new}}$ cheaper for $q_{\text{near}}$, change parent.
\end{enumerate}
$\Rightarrow$ probabilistic completeness + \sub{asymptotic optimality}.

\sect{8. Constraints \& Nonholonomy \hfill {\scriptsize[L04]}}
\sub{Pfaffian}: $A(q)\dot q=0$, $k$ constraints, $n-k$ DoF velocity.
\sub{Holonomic} (integrable): can write as $h(q)=0$ $\Rightarrow$ lose dimension. \sub{Nonholonomic}: not integrable; velocity constrained but reachability preserved.

\sub{Simple car}: $\dot x=v\cos\theta$, $\dot y=v\sin\theta$, $\dot\theta=v\tan\phi/L$, Pfaffian $\dot x\sin\theta-\dot y\cos\theta=0$.

Control-affine driftless: $\dot q=\sum g_i(q)u_i = G(q)u$.

\sub{Lie bracket}: $[f,g]=\tfrac{\partial g}{\partial x}f-\tfrac{\partial f}{\partial x}g$. Generates new feasible directions over time.

\sub{Frobenius}: $\mathcal{D}=\text{span}\{g_i\}$ involutive ($[g_i,g_j]\in\mathcal{D}$) $\iff$ holonomic.

\sub{Rank test}: if rank$[g_1\dots g_m\ [g_i,g_j]]>m$ for some $(i,j)$ → nonholonomic.

\sect{9. Controllability \hfill {\scriptsize[L04,05]}}
\sub{Pfaffian $\to$ control system}: $H(q)\dot q=0$ ($k$ rows). Null space basis $\{g_1,\dots,g_{n-k}\}$ $\Rightarrow$ $\dot q=G(q)u$, $m=n-k$ inputs. Basis choice 비유일 (다른 $u$ 의미).

\sub{Controllable}: $\forall q_i,q_f\in\mathcal C$, $\exists u(t)$ steering $q_i\to q_f$. 제약이 도달성에 영향 X.

$\Delta_{\mathcal A}=\mathcal L(\Delta)$ = involutive closure / \sub{Lie algebra} of $\Delta=\text{span}\{g_1,\dots,g_m\}$ = span of all nested Lie brackets:
$$\mathcal L(\Delta)=\text{span}\{g_i,\,[g_i,g_j],\,[g_i,[g_j,g_k]],\dots\}$$
$\Delta_i=\Delta_{i-1}+\text{span}\{[g,v]:g\in\Delta_1,v\in\Delta_{i-1}\}$. Accessible dim $\nu=\dim\mathcal L(\Delta)$.

\sub{왜 nested?}: 1단 bracket으론 부족할 수 있음.
\begin{itemize}
\item Unicycle (n=3,m=2): 1단 $[g_1,g_2]$로 충분.
\item Chained-form (n=5,m=2): 1단 rank 3, 2단 4, 3단 5 — 3단까지 필요.
\item Snakeboard: drift와 $[f,g_i]$, $[f,[f,g_i]]$ 등 더 깊이.
\end{itemize}

\sub{Termination}: 새 bracket이 rank 못 늘리면 stop. $\dim\mathcal L\le n$에서 포화. ``No additional vector fields can possibly be independent.''

\sub{Chow's theorem}: $\dim\mathcal L(\Delta)(x)=n\,\forall x \iff$ controllable (driftless).
Frobenius (1단)로 holonomic 판정 ↔ Chow (전체 nested)로 controllability 판정.

\sub{3-way 분류} ($p$ = integrable combos, $\nu=n-p$):
\begin{tabular}{@{}lccc@{}}
\toprule
case & $\nu$ vs $n,m$ & $p$ & 성질 \\
\midrule
완전 nonhol & $\nu=n$ & $p=0$ & controllable, STLC \\
부분 적분 & $m<\nu<n$ & $0<p<k$ & still nonhol \\
완전 holonomic & $\nu=m$ & $p=k$ & integrable \\
\bottomrule
\end{tabular}

\sub{Chow–Rashevskii} (driftless): $\dim\mathcal{L}\{g_1,\dots,g_m\}=n \Rightarrow$ STLC. Accessibility $\iff$ controllability.

With drift: accessibility $\not\Rightarrow$ controllability.

Degree of nonholonomy $\kappa\le n-m+1$.

\sect{10. Dubins / Reeds-Shepp \hfill {\scriptsize[L04]}}
\sub{Dubins} (forward only, min radius $\rho$): optimal $\in\{LRL, RLR, LSL, LSR, RSL, RSR\}$, $\alpha,\gamma\in[0,2\pi)$, $\beta\in(\pi,2\pi)$, $d\ge 0$. CCC if $|q_g-q_i|<4\rho$.

\sub{Reeds-Shepp} (forward+reverse): 48 base words including $C|C|C$, $CC|C$, $CSC$, etc.

\sect{11. Chained Form \hfill {\scriptsize[L05]}}
2-input, $n$-state:
$\dot z_1=u_1,\ \dot z_2=u_2,\ \dot z_3=z_2 u_1,\dots,\dot z_n=z_{n-1}u_1$.
\sub{Sufficient conditions}: $\Delta_0=\text{span}\{g_1,g_2,\text{ad}_{g_1}g_2,\dots,\text{ad}_{g_1}^{n-2}g_2\}$ has dim $n$; $\Delta_1, \Delta_2$ involutive; $\exists h_1$ with $dh_1\Delta_1=0$, $dh_1 g_1=1$; $\exists h_2$ indep of $h_1$ with $dh_2 \Delta_2=0$. Then $z_1=h_1$, $z_k=L_{g_1}^{n-k}h_2$.

\sub{Unicycle}: $z_1=\theta$, $z_2=x\cos\theta+y\sin\theta$, $z_3=x\sin\theta-y\cos\theta$. $\omega=v_1$, $v=v_2+z_3 v_1$.

\sub{Bicycle (RWD)}: $z_1=x$, $z_2=\sec^3\theta\tan\phi/L$, $z_3=\tan\theta$, $z_4=y$. $v=v_1/\cos\theta$, $\omega=-\tfrac{3}{L}v_1\sec\theta\sin^2\phi+\tfrac{1}{L}v_2\cos^3\theta\cos^2\phi$. Singular at $\cos\theta=0$.

\sect{12. Differential Flatness \hfill {\scriptsize[L05,06,09]}}
$\exists$ flat output $y$ s.t. $x=x(y,\dot y,\dots,y^{(r)})$, $u=u(y,\dot y,\dots,y^{(r)})$.
Driftless: \sub{flatness $\iff$ chained form transform}; flat outputs are $z_1, z_n$.
\sub{Unicycle/bicycle}: $(x,y)$ flat.
$\theta=\arctan(\dot y/\dot x)$, $v=\pm\sqrt{\dot x^2+\dot y^2}$, $\omega=(\ddot y\dot x-\ddot x\dot y)/(\dot x^2+\dot y^2)$.
\sub{Quadrotor}: $(x,y,z,\psi)$ flat. $f=m\|a\|$, $\mathbf b_3=a/\|a\|$ with $a=g\mathbf e_3-\ddot{\mathbf p}$. Snap → moments.

\sect{13. Polynomial Trajectory \hfill {\scriptsize[L06]}}
Cost $J=\int_0^T (P^{(r)}(t))^2 dt$.
Euler–Lagrange: $\sum_{k=0}^r(-1)^k\tfrac{d^k}{dt^k}\tfrac{\partial L}{\partial x^{(k)}}=0$.
Optimal $P$: degree $2r-1$, requires $2r$ BCs.
\begin{itemize}
\item $r=1$ min vel: cubic
\item $r=2$ min accel: quintic
\item $r=3$ min jerk: septic
\item $r=4$ min snap: 9th order
\end{itemize}

QP form: $\min p^T Q p$ s.t. $Ap=b$. With $T^{i+k-2r+1}/(i+k-2r+1)$ entries.

\sub{Multi-segment}: enforce $C^0,\dots,C^{N-1}$ at waypoints, $[A_T^i\ -A_0^{i+1}][p^i;p^{i+1}]=0$. Time alloc: $J_{\text{aug}}=J+\sum c_k T_k$.

\sub{Quintic Hermite (5th order)}: $p(u)=10u^3-15u^4+6u^5$ for unit BC.
\sub{Min jerk (7th)}: $35u^4-84u^5+70u^6-20u^7$.

\sect{14. Bézier / Bernstein \hfill {\scriptsize[L06,09]}}
$B_{k,N}(t)=\binom{N}{k}t^k(1-t)^{N-k}$, $r(t)=\sum b_k B_{k,N}$.
\begin{itemize}
\item \sub{Endpoint interpolation}: $r(0)=b_0$, $r(1)=b_N$.
\item \sub{Convex hull}: $r(t)\in\text{conv}\{b_k\}$ — safety: constrain control points!
\item \sub{Hodograph}: $r'$ is Bézier degree $N-1$, control points $b_k^{(1)}=N(b_{k+1}-b_k)$. Velocity/accel bounds = bounds on finite diffs of $b_k$.
\item \sub{Variation diminishing}: curve crosses line ≤ control polygon crosses.
\item \sub{Degree elevation}: $\tilde b_k=\tfrac{k}{N+1}b_{k-1}+(1-\tfrac{k}{N+1})b_k$.
\item \sub{de Casteljau}: $b_i^k=(1-t)b_{i-1}^{k-1}+t b_i^{k-1}$. Evaluate + subdivide.
\end{itemize}

$C^0$: $P_3=Q_0$.
$C^1$: $Q_1=2P_3-P_2$.
$C^2$: $Q_2=4P_3-4P_2+P_1$.

\sect{15. Collision Checking \hfill {\scriptsize[L07]}}
\sub{Segment intersection} (2D): orient$(A,B,C)=(B_x-A_x)(C_y-A_y)-(B_y-A_y)(C_x-A_x)$. Two segments cross iff orient signs on opposite sides for both pairs.

\sub{SAT} (convex polygons): for each edge normal $u$, compute projection intervals; if any separating axis → no collision.

\sub{GJK}: $A\cap B\ne\emptyset \iff 0\in A-B$. Build simplex via support: $\text{support}_{A-B}(d)=\text{support}_A(d)-\text{support}_B(-d)$. Iteratively add support points toward origin; if simplex contains 0 → collision.

\sub{EPA} (after GJK collision): expand polytope to find closest boundary point. Penetration depth $d$ + normal $n$. Recover contact: $v_i=a_i-b_i$, $p_A=\sum\lambda_i a_i$, $p_B=\sum\lambda_i b_i$, $p_A=p_B+dn$.

\sub{SDF} $\phi(x)$: signed distance (negative inside). Precompute on grid; query forward kinematics $x(q,u)$. Robot $\approx$ sphere: clearance $=\phi(x)-r$.

\sect{16. CHOMP \hfill {\scriptsize[L08]}}
$\min U(\xi)=f_{\text{prior}}(\xi)+f_{\text{obs}}(\xi)$, $\xi=(q_1,\dots,q_n)$.

\sub{Covariant update}: $\xi\leftarrow\xi-\tfrac1\lambda M^{-1}\nabla U$. $M$ = smoothness metric (PSD).

\sub{Smoothness}: $f_{\text{prior}}=\tfrac12\sum w_d\|K_d\xi+e_d\|^2$ where $K_d$ = $d$-th order finite difference. $=\tfrac12\xi^T A\xi+\xi^T b+c$.

\sub{Obstacle gradient} (workspace): $\tfrac{\delta J}{\delta x}=\|x'\|[(I-\hat x'\hat{x}'^T)\nabla c - c(x)\kappa]$.
Project to C-space: $\nabla_q f=J_{\text{robot}}^T \nabla_x f$.

Curvature: $\kappa=\tfrac{1}{\|x'\|^2}(I-\hat x'\hat{x}'^T)x''$.

Obstacle cost: $c(x)=\tfrac{1}{2\epsilon}(d(x)-\epsilon)^2$ for $0\le d\le\epsilon$, $-d+\tfrac\epsilon 2$ if $d<0$, else 0.

\sub{HMC inside CHOMP}: $p(\xi)\propto\exp(-\alpha U)$, momentum $\gamma$, Hamiltonian $H=U+\tfrac12\gamma^T\gamma$. Escape local minima.

\sect{17. Safe Flight Corridor \hfill {\scriptsize[L09]}}
Given path $\{p_i\to p_{i+1}\}$:
\begin{enumerate}
\item \sub{Ellipsoid} around segment $L_i$: start sphere, find nearest obstacle $p^*$, shrink to touch, repeat.
\item \sub{Polyhedron}: tangent hyperplanes at obstacle points. $a_j=2E^{-1}E^{-T}(p_j^c-d)$, $b_j=a_j^T p_j^c$. $C_i=\bigcap H_j$.
\item Optimize Bézier with control points in $C_i$ (convex hull → safety).
\end{enumerate}

\sub{FMM} (Fast Marching): $|\nabla T|=1/f(x)$ (Eikonal). $f(d)$ from SDF: low near obstacles. $O(N\log N)$. Path: gradient descent on $T$.

\sub{Cube corridor}: axis-aligned inflated cubes — linear constraints $A_i p\le b_i$.

\sect{18. LQR \& Riccati \hfill {\scriptsize[L10]}}
$x_{k+1}=A_k x_k+B_k u_k$, $J=\tfrac12\sum x^T Q x+u^T R u + \tfrac12 x_N^T Q_N x_N$.
$V_k(x)=\tfrac12 x^T P_k x$.
\[K_k=(R+B^T P_{k+1}B)^{-1}B^T P_{k+1}A\]
\[P_k=Q+A^T P_{k+1}A-A^T P_{k+1}B(R+B^T P_{k+1}B)^{-1}B^T P_{k+1}A\]
$P_N=Q_N$. Time-varying linear feedback $u_k^*=-K_k x_k$.

\sect{19. iLQR / DDP \hfill {\scriptsize[L10,12]}}
Nominal $\{\bar x_k,\bar u_k\}$. Linearize $\delta x_{k+1}=A_k\delta x_k+B_k\delta u_k$.
\sub{Q-function}:
$Q_x=\ell_x+A^T V_x$, $Q_u=\ell_u+B^T V_x$
$Q_{xx}=\ell_{xx}+A^T V_{xx}A$, $Q_{uu}=\ell_{uu}+B^T V_{xx}B$
$Q_{ux}=\ell_{ux}+B^T V_{xx}A$.

\sub{Local policy}: $\delta u^*=k+K\delta x$, $k=-Q_{uu}^{-1}Q_u$, $K=-Q_{uu}^{-1}Q_{ux}$.

\sub{Value update}:
$V_x=Q_x-Q_{ux}^T Q_{uu}^{-1}Q_u$
$V_{xx}=Q_{xx}-Q_{ux}^T Q_{uu}^{-1}Q_{ux}$.

\sub{Forward rollout}: $u_k^{\text{new}}=\bar u_k+\alpha k_k+K_k(x_k^{\text{new}}-\bar x_k)$, line search $\alpha\in(0,1]$.

\sub{Regularization}: $Q_{uu}^{\text{reg}}=Q_{uu}+\lambda I$.

\sub{DDP}: includes $f_{xx},f_{ux},f_{uu}$ — extra terms $V_{x,k+1}\cdot F_{**}$ in $Q_{**}$. iLQR drops them (Gauss-Newton).

\sect{20. Linear MPC \hfill {\scriptsize[L11]}}
$\min\sum_{i=0}^{N-1}(x^T Q x+u^T R u)+x_N^T P x_N$ s.t. dyn + $x\in\mathcal X, u\in\mathcal U, x_N\in\mathcal X_f$. Apply only $u_0^*$, shift, repeat.

\sub{Condensed QP}: $X=\Phi x_t+\Gamma U$, $J=U^T HU+2x^T F^T U+\text{const}$, $H=\Gamma^T\bar Q\Gamma+\bar R$. Small variable, dense.

\sub{Sparse QP}: $Z=[x_0;u_0;\dots;x_N]$, block-banded $A_{eq}$. Larger but exploits sparsity.

\sub{KKT}: $\begin{pmatrix}H&A^T\\A&0\end{pmatrix}\begin{pmatrix}Z\\\lambda\end{pmatrix}=\begin{pmatrix}-q\\b\end{pmatrix}$.

\sect{21. MPC Stability \hfill {\scriptsize[L11]}}
\sub{Terminal cost} $V_f$ approximates $V_\infty$. Choose $K=K_{\text{LQR}}$, $A_K=A+BK$, solve DARE: $P=A^T PA-A^T PB(R+B^T PB)^{-1}B^T PA+Q$.

\sub{Terminal set} $\mathcal X_f$: invariant under $K$, $K\mathcal X_f\subseteq\mathcal U$, $\mathcal X_f\subseteq\mathcal X$, $V_f(A_K x)-V_f(x)\le-\ell(x,Kx)$.

\sub{Recursive feasibility}: shift optimal sequence + append $Kx_N^*$ remains feasible.

\sub{Lyapunov decrease}: $V_N(x_{t+1})-V_N(x_t)\le-\ell(x_t,\kappa_N(x_t))$. Sum to $\infty$ → $x_t\to 0$.

\sub{Pitfall}: terminal cost alone insufficient; need both cost+set.

Soft constraint: $C x\le d+s$, $s\ge 0$, penalty $\rho_1\mathbf{1}^T s+\rho_2 s^T s$.

\sect{22. NMPC \& PMP \hfill {\scriptsize[L12]}}
\sub{Lagrangian}:
$\mathcal L=\sum\ell+\ell_f+\sum\lambda_k^T(x_{k+1}-F(x_k,u_k))$.

\sub{Costate} (backward): $\lambda_N=\ell_{f,x}$, $\lambda_k=\ell_x+F_x^T\lambda_{k+1}$.

\sub{Optimality}: $\ell_u+F_u^T\lambda_{k+1}=0 \Rightarrow u_k^*=\arg\min_u[\ell+\lambda_{k+1}^T F]$.

\sub{SQP}: linearize + quadratic cost → QP for $(\delta x,\delta u)$, $\delta x_{k+1}=A_k\delta x_k+B_k\delta u_k$, update $\bar u_k\leftarrow\bar u_k+\alpha\delta u_k$.

\sub{Continuous PMP}: $H(x,u,\lambda)=\ell+\lambda^T f$, $\dot x=\partial_\lambda H$, $\dot\lambda=-\partial_x H$, $u^*=\arg\min_u H$, $\lambda(T)=\partial_x\ell_f$. Two-point BVP (shooting).

\sub{Discrete↔PMP}: $\lambda_k$ discrete = $\lambda(t)\Delta t$ continuous, costate recursion matches $\dot\lambda=-\partial_x H$ as $\Delta t\to 0$.

\sect{23. Stability (discrete) \hfill {\scriptsize[L11]}}
$V(x)>0$, $V(0)=0$, $\Delta V=V(f(x))-V(x)$.
\begin{itemize}
\item Lyapunov: $\Delta V\le 0$
\item Asymptotic: $\Delta V\le -\ell(x)$, $\ell>0$
\item Exponential: $\Delta V\le -\alpha V$
\end{itemize}
LTI $x_{k+1}=Ax_k$, $\rho(A)<1$: $V=x^T Px$ where $A^T PA-P=-Q$, $Q>0$.

\sub{Key formulas to remember}:
\begin{itemize}
\item Pfaffian for car: $\sin\theta\dot x-\cos\theta\dot y=0$
\item Curvature: $\kappa=(\dot x\ddot y-\dot y\ddot x)/(\dot x^2+\dot y^2)^{3/2}$
\item Min-jerk BC: pos+vel+accel+jerk=0 each end
\item DARE: $P=A^T PA-A^T PB(R+B^T PB)^{-1}B^T PA+Q$
\item iLQR gain: $K=-Q_{uu}^{-1}Q_{ux}$, $k=-Q_{uu}^{-1}Q_u$
\item Bézier hodograph: $b_k^{(1)}=N(b_{k+1}-b_k)$
\item Hamiltonian: $H=\ell+\lambda^T f$, $\dot\lambda=-H_x$
\end{itemize}
